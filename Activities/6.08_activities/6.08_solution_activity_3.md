Sub macro_loop_ifElse()

Dim eid As String
eid = "EMP002"
Dim name As String
name = "John Doe"
Dim age As String
age = "60"
Dim email As String
email = "john.doe@ironhack.com"
Dim dob As Date
dob = Format("4/22/1980", "Short Date")
Dim married As Boolean
married = True
Dim location As String
location = "Orlando"

Range("A1").Select
Dim x As String
x = LCase(Range("A1").Value)

Do
    If x = "name" Then
    Selection.Offset(2, 0).Select
    Selection.Value = name
    Selection.Offset(-2, 1).Select

    ElseIf x = "eid" Then
    Selection.Offset(2, 0).Select
    Selection.Value = eid
    Selection.Offset(-2, 1).Select

    ElseIf x = "age" Then
    Selection.Offset(2, 0).Select
    Selection.Value = age
    Selection.Offset(-2, 1).Select

    ElseIf x = "email" Then
    Selection.Offset(2, 0).Select
    Selection.Value = email
    Selection.Offset(-2, 1).Select

    ElseIf x = "dob" Then
    Selection.Offset(2, 0).Select
    Selection.Value = married
    Selection.Offset(-2, 1).Select

    ElseIf x = "location" Then
    Selection.Offset(2, 0).Select
    Selection.Value = location
    Selection.Offset(-2, 1).Select

    End If
    x = LCase(Selection.Value)
Loop While IsEmpty(Selection) = False

End Sub
