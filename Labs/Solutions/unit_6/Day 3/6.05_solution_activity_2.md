```sql
drop procedure if exists total_loss_in_district_var;

delimiter //
create procedure total_loss_in_district_var (in param0 text)
begin
  declare district_loss float default 0.0;
  select round((sum(l.amount) - sum(l.payments)), 2) as loss INTO district_loss
  from account a
  join district d
  on d.A1 = a.district_id
  join loan l
  on l.account_id = a.account_id
  where l.status = "B" and binary d.A2 = binary param0;
  select district_loss;
  set district_loss = 0.0;
end;
//
delimiter ;

call total_loss_in_district_var("Kutna Hora");
```
