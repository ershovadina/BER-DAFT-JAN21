```sql
drop procedure if exists total_loss_in_district;
delimiter //
create procedure total_loss_in_district (in param0 text, out param1 float)
begin
  select round((sum(l.amount) - sum(l.payments)), 2) as loss into param1
  from account a
  join district d
  on d.A1 = a.district_id
  join loan l
  on l.account_id = a.account_id
  where l.status = "B" and binary d.A2 = binary param0;
end;
//
delimiter ;
call total_loss_in_district("Kutna Hora", @x);
select @x as "Kutna Hora loss";
```
